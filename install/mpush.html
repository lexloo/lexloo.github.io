            # 安装JRE >1.8

            # 安装redis

            yum install redis

#如果查找不到redis
yum install epel-release

            # 安装zookeeper

            wget http://mirror.bit.edu.cn/apache/zookeeper/zookeeper-3.4.14/zookeeper-3.4.14.tar.gz

tar -zxvf zookeeper-3.4.14.tar.gz
cd zookeeper-3.4.14
cp conf/zoo_sample.cfg conf/zoo.cfg

bin/zkServer.sh start

#查看
netstat -tlnp | grep 2181

            # 安装mpush

            wget https://github.com/mpusher/mpush/releases/download/v0.7.0/mpush-release-0.7.0.tar.gz
tar -zxvf mpush-release-0.7.0.tar.gz
cd mpush-release-0.7.0
chmod +x bin/*.sh
bin/mp.sh start
# 查看端口3000,3001,3002 是否启动

            # 安装alloc

            wget https://github.com/mpusher/alloc/releases/download/v0.7.0/alloc-release-0.7.0.tar.gz
tar -zxvf alloc-release-0.7.0
cd alloc-release-0.7.0
vim conf/mpush.conf
chmod +x bin/*.sh
bin/mp.sh start
